// -- TAKE FIRST INPUT--
FST, SKI
    BUN FST
    CLA
    INP
    STA TWOASCII

// -- TAKE SECOND INPUT --
SCD, SKI
    BUN SCD
    CLA
    INP
    STA ONEASCII

// -- CHECK IF FIRST INPUT IS A NUMBER --
CLA
LDA ONEASCII
ADD HBOUND
SNA
BUN FLETTER
BUN FNUMBER

FNUMBER, CLA
    LDA ONEASCII
    ADD DECTODEC
    STA ONEDEC
    BUN SECINP

FLETTER, CLA
    LDA ONEASCII
    ADD LETTODEC
    STA ONEDEC
    BUN SECINP

// -- CHECK IF SECOND INPUT IS A NUMBER --
SECINP, CLA
    LDA TWOASCII
    ADD HBOUND
    SNA
    BUN SLETTER
    BUN SNUMBER

SNUMBER, CLA
    LDA TWOASCII
    ADD DECTODEC
    STA TWODEC
    BUN ADDER

SLETTER, CLA
    LDA TWOASCII
    ADD LETTODEC
    STA TWODEC
    BUN ADDER

// -- CONVERT HEX TO DEC --
ADDER, CLA
    LDA ONEDEC
    STA DECINP

MULSTART, CLA
    LDA MULCOUNT
    ADD SIXTEEN
    SNA
    BUN LARGEST
    CLA
    LDA DECINP
    ADD TWODEC
    STA DECINP
    CLA
    LDA MULCOUNT
    INC
    STA MULCOUNT
    BUN MULSTART

// -- FIND LARGEST DIGIT --
LARGEST, CLA
    LDA DECINP
    SZA
    BUN OUTPUT
    SNA
    BUN OUTPUT

    CLA
    LDA DECINP
    STA DIVIDEND
    CLA
    LDA ZERO
    STA QUOTIENT
    CLA
    LDA ZERO
    STA CURRDEC

    CHKDIV, CLA
        LDA DIVIDEND
        ADD TEN
        SNA
        BUN DIVIDER
        SZA
        BUN DIVIDER
    
    CLA
    LDA DIVIDEND
    STA CURRDEC
    OUT

    CLA
    LDA LARGESTDEC
    CMA
    ADD ONE
    STA CLARGESTDEC

    CLA
    LDA CURRDEC
    ADD CLARGESTDEC
    SPA
    BUN NOREPLACE
    SZA
    BUN NOREPLACE

    CLA
    LDA CURRDEC
    STA LARGESTDEC

    NOREPLACE, CLA

    CLA
    LDA QUOTIENT
    STA DECINP

    BUN LARGEST

DIVIDER, CLA
    LDA DIVIDEND
    ADD TEN
    STA DIVIDEND
    CLA
    LDA QUOTIENT
    ADD ONE
    STA QUOTIENT
    BUN CHKDIV

// -- OUTPUT --
OUTPUT, CLA
    LDA LARGESTDEC
    ADD DECTODECP
    OUT

// --================--
// -- END OF PROGRAM --
// --================--

// -- VARIABLES --
// INPUT ASCII
ONEASCII, DEC 0
TWOASCII, DEC 0

// INPUT HEX AS DECIMAL
ONEDEC, DEC 0
TWODEC, DEC 0

// INPUT DECIMAL
DECINP, DEC 0

// COMPARITORS FROM ASCII
HBOUND, DEC -58
LBOUND, DEC -47

// CONVERTERS FROM ASCII
DECTODEC, DEC -48
DECTODECP, DEC 48
LETTODEC, DEC -55

// LARGEST DIGIT STORAGE
LARGESTDEC, DEC 0
CLARGESTDEC, DEC 0

// LOOPERS
SIXTEEN, DEC -16
MULCOUNT, DEC 0
ONE, DEC 1

// MORE LOOPERS
TEN, DEC -10
NINE, DEC -9
DIVIDEND, DEC 0
QUOTIENT, DEC 0

CURRDEC, DEC 0

// RANDOMS
ZERO, DEC 0

// -- END OF CODE --
END
