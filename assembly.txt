// -- TAKE FIRST INPUT--
FST, SKI
    BUN FST
    CLA
    INP
    STA TWOASCII

// -- TAKE SECOND INPUT --
SCD, SKI
    BUN SCD
    CLA
    INP
    STA ONEASCII

// -- CHECK IF FIRST INPUT IS A NUMBER --
CLA
LDA ONEASCII
ADD HBOUND
SNA
BUN FLETTER
BUN FNUMBER

FNUMBER, CLA
    LDA ONEASCII
    ADD DECTODEC
    STA ONEDEC
    BUN SECINP

FLETTER, CLA
    LDA ONEASCII
    ADD LETTODEC
    STA ONEDEC
    BUN SECINP

// -- CHECK IF SECOND INPUT IS A NUMBER --
SECINP, CLA
    LDA TWOASCII
    ADD HBOUND
    SNA
    BUN SLETTER
    BUN SNUMBER

SNUMBER, CLA
    LDA TWOASCII
    ADD DECTODEC
    STA TWODEC
    BUN ADDER

SLETTER, CLA
    LDA TWOASCII
    ADD LETTODEC
    STA TWODEC
    BUN ADDER

// -- CONVERT HEX TO DEC --
ADDER, CLA
    LDA ONEDEC
    STA DECINP

MULSTART, CLA
    LDA MULCOUNT
    ADD SIXTEEN
    SNA
    BUN LARGEST
    CLA
    LDA DECINP
    ADD TWODEC
    STA DECINP
    CLA
    LDA MULCOUNT
    INC
    STA MULCOUNT
    BUN MULSTART

// -- FIND LARGEST DIGIT --
LARGEST, CLA
    LDA DECINP
    SPA
    BUN OUTPUT
    BUN DECLOOP

DECLOOP, CLA
    LDA ZERO
    STA QUOTIENT
    CLA
    LDA DECINP
    STA DIVIDEND

DIVCHK, CLA
    LDA DIVIDEND
    ADD NINE
    SPA
    BUN DECEND
    BUN DIVLOOP

DIVLOOP, CLA
    LDA DIVIDEND
    ADD TEN
    STA DIVIDEND
    CLA
    LDA QUOTIENT
    INC
    STA QUOTIENT
    BUN DIVCHK

DECEND, CLA

// -- OUTPUT --
OUTPUT, CLA
    LDA DIVIDEND
    ADD DECTODECP
    OUT

// --================--
// -- END OF PROGRAM --
// --================--

// -- VARIABLES --
// INPUT ASCII
ONEASCII, DEC 0
TWOASCII, DEC 0

// INPUT HEX AS DECIMAL
ONEDEC, DEC 0
TWODEC, DEC 0

// INPUT DECIMAL
DECINP, DEC 0

// COMPARITORS FROM ASCII
HBOUND, DEC -58
LBOUND, DEC -47

// CONVERTERS FROM ASCII
DECTODEC, DEC -48
DECTODECP, DEC 48
LETTODEC, DEC -55

// LARGEST DIGIT STORAGE
LARGESTDEC, DEC 0
CLARGESTDEC, DEC 0

// LOOPERS
SIXTEEN, DEC -16
MULCOUNT, DEC 0
ONE, DEC 1
NONE, DEC -1

// MORE LOOPERS
TEN, DEC -10
NINE, DEC -9
DIVIDEND, DEC 0
QUOTIENT, DEC 0

CURRDEC, DEC 0

// RANDOMS
ZERO, DEC 0
PLUSSIGN, DEC 43

// -- END OF CODE --
END
